public boolean isColunaComZero(double[] col, int lin) throws Exception 
{
	if(col == null)
		throw new Exception("coluna para validar inexistente");

	for(int i = 0; i < col.length; i++)
	{
		if(i == lin)
			i++;
				
		
		if(col[i] != 0)
			return false;
	}

	return true;
}
